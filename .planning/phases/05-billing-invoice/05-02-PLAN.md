    ```typescript
    @Controller('billing-runs')
    export class BillingSseController {
      constructor(private readonly eventEmitter: EventEmitter2) {}

      @Sse(':id/progress')
      @Public()
      progress(@Param('id') billingRunId: string, @Query('token') token: string): Observable<MessageEvent> {
        // Optional: validate token if provided (basic security for SSE)
        // For demo, accept all connections

        const timeout$ = timer(5 * 60 * 1000); // 5 min max connection

        return fromEvent(this.eventEmitter, 'billing.progress').pipe(
          filter((event: BillingRunProgressEvent) => event.billingRunId === billingRunId),
          map((event: BillingRunProgressEvent) => ({
            data: JSON.stringify(event),
          }) as MessageEvent),
          takeUntil(timeout$),
          finalize(() => {
            // Cleanup on disconnect
          }),
        );
      }
    }
    ```

    Import from rxjs: `Observable, fromEvent, map, filter, takeUntil, timer, finalize`.
    Import `MessageEvent` from `@nestjs/common`.
    Use `@Sse()` decorator from `@nestjs/common`.
    Use `@Public()` decorator to bypass JWT (SSE EventSource cannot send headers).

    **Step 3: Add methods to billing.service.ts**

    `cancelTenants(billingRunId, dto: CancelTenantDto)`:
    1. Load billing run, verify NOT in terminal state
    2. Get all obligations linked to this billing run AND matching dto.tenantIds
    3. Set those obligations' billingRunId = null (unlink), keep their status as 'ready'
    4. Update billing run filters: add cancelled tenantIds to filters.cancelledTenants
    5. Check if ALL tenants are now cancelled (compare filters.tenantIds vs filters.cancelledTenants). If all cancelled, transition run to 'cancelled'.
    6. Emit 'billing.progress' event with updated tenant progress

    `rerunBillingRun(dto: RerunBillingRunDto)`:
    1. Load previous run, verify it IS in a terminal state (completed, cancelled, partial, rejected)
    2. Determine run mode:
       - If previousRun.status === 'cancelled' -> runMode = 'full'
       - If previousRun.status === 'completed' -> runMode = 'delta'
       - If previousRun.status === 'partial' -> runMode = 'delta' (re-run only failed tenants)
    3. Create new billing run with same airportId, periodStart, periodEnd, filters from previous
       Set previousRunId = dto.previousRunId, runMode as determined
    4. Add to billing-run queue
    5. Return new run
