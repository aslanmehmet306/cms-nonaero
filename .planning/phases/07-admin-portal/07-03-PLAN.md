// InvoicesController — /api/v1/invoices
GET    /invoices                   // query: ?status, ?tenantId, ?page, ?limit
GET    /invoices/:id
// Each invoice record has: id, stripeInvoiceId, stripeInvoiceUrl, status, amount, currency, tenantId, billingRunId, periodStart, periodEnd, issuedAt, dueDate, paidAt

// ReportsController — /api/v1/reports
GET    /reports/dashboard          // query: ?airportId, ?currency
// Returns: { totalRevenue, outstandingInvoices, collectionRate, activeTenants, activeContracts, pendingObligations }
GET    /reports/revenue-summary    // query: ?airportId, ?periodStart, ?periodEnd, ?groupBy (tenant|service_type), ?currency
// Returns: { items: [{ key, label, amount, currency }], total, warning? }
GET    /reports/aging              // query: ?airportId, ?asOfDate, ?currency
// Returns: { current, days30, days60, days90, days90plus, total }
GET    /reports/obligations        // query: ?airportId, ?tenantId, ?status, ?periodStart, ?periodEnd, ?page, ?limit
GET    /reports/billing-history    // query: ?airportId, ?page, ?limit

// BillingPoliciesController — /api/v1/billing-policies
GET    /billing-policies           // query: ?airportId
GET    /billing-policies/:id
POST   /billing-policies           // { airportId, name, cutoffDay, issueDay, dueDateDays, fiscalYearStartMonth, fiscalYearStartDay }
PATCH  /billing-policies/:id
POST   /billing-policies/:id/activate

// UsersController — /api/v1/users
GET    /users                      // query: ?role, ?page, ?limit
GET    /users/:id
POST   /users                      // { email, password, name, role, airportId? }
PATCH  /users/:id

// AirportsController — /api/v1/airports
GET    /airports
GET    /airports/:id
PATCH  /airports/:id
