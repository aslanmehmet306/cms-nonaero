    ```typescript
    @Injectable()
    export class SettlementService {
      constructor(private readonly prisma: PrismaService) {}

      async calculateMonthlyMag(
        contractId: string,
        periodStart: Date,
        periodEnd: Date,
      ): Promise<void> {
        // a. Fetch contract with annualMag
        const contract = await this.prisma.contract.findUnique({ where: { id: contractId } });
        if (!contract || !contract.annualMag) return; // No MAG clause

        // b. Calculate monthly MAG = annualMag / 12
        const monthlyMag = DecimalHelper.divide(contract.annualMag.toString(), 12);

        // c. Find the revenue_share obligation for this period (status=ready)
        //    This is the obligation whose calculated amount represents the revenue share
        const revShareObligation = await this.prisma.obligation.findFirst({
          where: {
            contractId,
            periodStart,
            chargeType: ChargeType.revenue_share,
            status: ObligationStatus.ready,
          },
        });

        const revenueShareAmount = revShareObligation
          ? new Decimal(revShareObligation.amount!.toString())
          : new Decimal(0);

        // d. Calculate higher-of result and shortfall
        const higherOfResult = DecimalHelper.max(revenueShareAmount, monthlyMag);
        const shortfall = DecimalHelper.subtract(monthlyMag, revenueShareAmount);

        // e. Create SettlementEntry for audit trail
        await this.prisma.settlementEntry.create({
          data: {
            airportId: contract.airportId,
            contractId: contract.id,
            tenantId: contract.tenantId,
            periodStart,
            periodEnd,
            settlementType: SettlementType.monthly_mag,
            revenueShareAmount: DecimalHelper.roundMoney(revenueShareAmount),
            magAmount: DecimalHelper.roundMoney(monthlyMag),
            higherOfResult: DecimalHelper.roundMoney(higherOfResult),
            shortfall: shortfall.isPositive() ? DecimalHelper.roundMoney(shortfall) : new Decimal(0),
            surplus: shortfall.isNegative() ? DecimalHelper.roundMoney(shortfall.abs()) : new Decimal(0),
          },
        });

        // f. If shortfall > 0, create/update mag_shortfall obligation
        if (shortfall.isPositive()) {
          const lineHash = buildLineHash(contract.tenantId, periodStart, ChargeType.mag_settlement);
          const dueDate = new Date(periodEnd.getTime() + 30 * 86400000);

          await this.prisma.obligation.upsert({
            where: { lineHash },
            create: {
              airportId: contract.airportId,
              contractId: contract.id,
              contractVersion: contract.version,
              tenantId: contract.tenantId,
              serviceDefinitionId: null, // MAG has no service definition
              obligationType: ObligationType.mag_shortfall,
              chargeType: ChargeType.mag_settlement,
              periodStart,
              periodEnd,
              dueDate,
              amount: DecimalHelper.roundMoney(shortfall),
              currency: contract.magCurrency ?? 'TRY',
              status: ObligationStatus.ready, // MAG shortfall is auto-ready
              lineHash,
              calculationTrace: {
                type: 'mag_settlement',
                monthlyMag: monthlyMag.toString(),
                revenueShareAmount: revenueShareAmount.toString(),
                shortfall: shortfall.toString(),
                formula: 'max(revenue_share, annual_mag / 12) - revenue_share',
              },
            },
            update: {
              amount: DecimalHelper.roundMoney(shortfall),
              calculationTrace: {
                type: 'mag_settlement_updated',
                monthlyMag: monthlyMag.toString(),
                revenueShareAmount: revenueShareAmount.toString(),
                shortfall: shortfall.toString(),
              },
            },
          });
        }
        // g. If shortfall <= 0: no obligation created (no carry-forward per R7.4)
      }

      async calculateYearEndTrueUp(
        contractId: string,
        fiscalYearStart: Date,
        fiscalYearEnd: Date,
      ): Promise<{ trueUpAmount: Decimal; created: boolean }> {
        // a. Fetch contract with annualMag
        const contract = await this.prisma.contract.findUnique({ where: { id: contractId } });
        if (!contract || !contract.annualMag) {
          return { trueUpAmount: new Decimal(0), created: false };
        }

        // b. Sum all revenue_share obligation amounts for the fiscal year
        const revShareObligations = await this.prisma.obligation.findMany({
          where: {
            contractId,
            chargeType: ChargeType.revenue_share,
            status: { in: [ObligationStatus.ready, ObligationStatus.invoiced, ObligationStatus.settled] },
            periodStart: { gte: fiscalYearStart },
            periodEnd: { lte: fiscalYearEnd },
          },
        });

        const annualRevenueShare = revShareObligations.reduce(
          (sum, o) => sum.plus(new Decimal(o.amount?.toString() ?? '0')),
          new Decimal(0),
        );

        const annualMag = new Decimal(contract.annualMag.toString());
        const trueUpAmount = DecimalHelper.subtract(annualMag, annualRevenueShare);

        // c. Also subtract any monthly mag_shortfall already billed
        const monthlyShortfalls = await this.prisma.obligation.findMany({
          where: {
            contractId,
            chargeType: ChargeType.mag_settlement,
            obligationType: ObligationType.mag_shortfall,
            status: { in: [ObligationStatus.ready, ObligationStatus.invoiced, ObligationStatus.settled] },
            periodStart: { gte: fiscalYearStart },
            periodEnd: { lte: fiscalYearEnd },
          },
        });

        const totalMonthlyShortfalls = monthlyShortfalls.reduce(
          (sum, o) => sum.plus(new Decimal(o.amount?.toString() ?? '0')),
          new Decimal(0),
        );

        // Net true-up = (annualMag - annualRevShare) - alreadyPaidShortfalls
        const netTrueUp = DecimalHelper.subtract(trueUpAmount, totalMonthlyShortfalls);

        // d. Create SettlementEntry
        await this.prisma.settlementEntry.create({
          data: {
            airportId: contract.airportId,
            contractId: contract.id,
            tenantId: contract.tenantId,
            periodStart: fiscalYearStart,
            periodEnd: fiscalYearEnd,
            settlementType: SettlementType.year_end_true_up,
            revenueShareAmount: DecimalHelper.roundMoney(annualRevenueShare),
            magAmount: DecimalHelper.roundMoney(annualMag),
            higherOfResult: DecimalHelper.roundMoney(DecimalHelper.max(annualRevenueShare, annualMag)),
            shortfall: netTrueUp.isPositive() ? DecimalHelper.roundMoney(netTrueUp) : new Decimal(0),
            surplus: netTrueUp.isNegative() ? DecimalHelper.roundMoney(netTrueUp.abs()) : new Decimal(0),
            trueUpAmount: netTrueUp.isPositive() ? DecimalHelper.roundMoney(netTrueUp) : new Decimal(0),
          },
        });

        // e. If net true-up > 0, create true-up obligation
        if (netTrueUp.isPositive()) {
          const lineHash = buildLineHash(contract.tenantId, fiscalYearStart, ChargeType.mag_settlement);
          const trueUpHash = createHash('sha256').update(`true_up:${lineHash}`).digest('hex');
          const dueDate = new Date(fiscalYearEnd.getTime() + 30 * 86400000);

          await this.prisma.obligation.create({
            data: {
              airportId: contract.airportId,
              contractId: contract.id,
              contractVersion: contract.version,
              tenantId: contract.tenantId,
              serviceDefinitionId: null,
              obligationType: ObligationType.mag_true_up,
              chargeType: ChargeType.mag_settlement,
              periodStart: fiscalYearStart,
              periodEnd: fiscalYearEnd,
              dueDate,
              amount: DecimalHelper.roundMoney(netTrueUp),
              currency: contract.magCurrency ?? 'TRY',
              status: ObligationStatus.ready,
              lineHash: trueUpHash,
              calculationTrace: {
                type: 'year_end_true_up',
                annualMag: annualMag.toString(),
                annualRevenueShare: annualRevenueShare.toString(),
                totalMonthlyShortfalls: totalMonthlyShortfalls.toString(),
                netTrueUp: netTrueUp.toString(),
              },
            },
          });

          return { trueUpAmount: DecimalHelper.roundMoney(netTrueUp), created: true };
        }

        return { trueUpAmount: new Decimal(0), created: false };
      }
    }
    ```

    ```typescript
    @OnEvent('obligation.calculated', { async: true })
    async handleObligationCalculated(event: ObligationCalculatedEvent): Promise<void> {

      if (event.chargeType !== ChargeType.revenue_share) return;
    ```
